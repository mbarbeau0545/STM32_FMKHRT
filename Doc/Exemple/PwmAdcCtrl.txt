#include "stm32g4xx_hal.h"

/* Déclarations globales */
HRTIM_HandleTypeDef hhrtim;
ADC_HandleTypeDef hadc1;

/* Prototypes de fonctions de configuration */
void SystemClock_Config(void);
static void MX_GPIO_Init(void);
static void MX_HRTIM1_Init(void);
static void MX_ADC1_Init(void);

int main(void)
{
    /* Initialisation de la librairie HAL et configuration du système */
    HAL_Init();
    SystemClock_Config();
    MX_GPIO_Init();
    MX_HRTIM1_Init();
    MX_ADC1_Init();

    /* Démarrer la génération PWM sur le Timer A du HRTIM */
    if (HAL_HRTIM_WaveformOutputStart(&hhrtim, HRTIM_OUTPUT_TA1) != HAL_OK)
    {
        /* Gestion d'erreur */
        Error_Handler();
    }
    /* Démarrer le compteur du Timer A */
    if (HAL_HRTIM_WaveformCounterStart(&hhrtim, HRTIM_TIMER_A) != HAL_OK)
    {
        Error_Handler();
    }
    
    /* Démarrer l'ADC en mode déclenché par le HRTIM */
    if (HAL_ADC_Start(&hadc1) != HAL_OK)
    {
        Error_Handler();
    }

    while (1)
    {
        /* Boucle principale, traitement de la conversion ou mise à jour de la régulation */
    }
}

/* Configuration du HRTIM1 : PWM + génération d'un événement trigger pour l'ADC */
static void MX_HRTIM1_Init(void)
{
    HRTIM_TimeBaseCfgTypeDef pTimeBaseCfg = {0};
    HRTIM_CompareCfgTypeDef pCompareCfg = {0};
    HRTIM_EventCfgTypeDef pEventCfg = {0};
    HRTIM_OutputCfgTypeDef pOutputCfg = {0};

    hhrtim.Instance = HRTIM1;
    hhrtim.Init.HRTIMInterruptResquests = HRTIM_IT_NONE;
    hhrtim.Init.SyncOptions = HRTIM_SYNCOPTION_NONE;
    if (HAL_HRTIM_Init(&hhrtim) != HAL_OK)
    {
        Error_Handler();
    }

    /* Configuration de la base de temps pour le Timer A
       Exemple : période définie pour obtenir la fréquence désirée (ici valeur fictive) */
    pTimeBaseCfg.Period = 50000;          // Valeur de période (ARR)
    pTimeBaseCfg.RepetitionCounter = 0;   // Pas de répétition supplémentaire
    pTimeBaseCfg.PrescalerRatio = HRTIM_PRESCALERRATIO_DIV1; // Ajuster si besoin
    pTimeBaseCfg.Mode = HRTIM_MODE_CONTINUOUS;
    if (HAL_HRTIM_WaveformTimeBaseConfig(&hhrtim, HRTIM_TIMER_A, &pTimeBaseCfg) != HAL_OK)
    {
        Error_Handler();
    }

    /* Configuration du comparateur pour générer l'événement trigger de l'ADC
       Ici, le trigger est positionné à la moitié de la période */
    pCompareCfg.CompareValue = 25000; // Déclenchement à mi-période
    if (HAL_HRTIM_WaveformCompareConfig(&hhrtim, HRTIM_TIMER_A, HRTIM_COMPARE_1, &pCompareCfg) != HAL_OK)
    {
        Error_Handler();
    }

    /* Configuration de l'événement du HRTIM pour l'ADC :
       Cet événement se produit sur le front montant de la comparaison */
    pEventCfg.Source = HRTIM_EVENTSRC_TIMER_A_CMP1;
    pEventCfg.Polarity = HRTIM_EVENTPOLARITY_RISING;
    pEventCfg.Sensitivity = HRTIM_EVENTSENSITIVITY_LEVEL;
    pEventCfg.FastMode = HRTIM_EVENTFASTMODE_ENABLE;
    if (HAL_HRTIM_EventConfig(&hhrtim, HRTIM_EVENT_1, &pEventCfg) != HAL_OK)
    {
        Error_Handler();
    }

    /* Configuration de la sortie PWM pour le Timer A, Channel 1 */
    pOutputCfg.Polarity = HRTIM_OUTPUTPOLARITY_HIGH;
    pOutputCfg.IdleMode = HRTIM_OUTPUTIDLEMODE_NONE;
    pOutputCfg.IdleLevel = HRTIM_OUTPUTIDLELEVEL_INACTIVE;
    pOutputCfg.FastMode = HRTIM_OUTPUTFASTMODE_DISABLE;
    pOutputCfg.ChopperModeEnable = HRTIM_OUTPUTCHOPPERMODE_DISABLE;
    pOutputCfg.DutyCycle = 50;  // Rapport cyclique à 50% (modifiable)
    if (HAL_HRTIM_WaveformOutputConfig(&hhrtim, HRTIM_TIMER_A, HRTIM_OUTPUT_TA1, &pOutputCfg) != HAL_OK)
    {
        Error_Handler();
    }
}

/* Configuration de l'ADC1 pour être déclenché par l'événement HRTIM */
static void MX_ADC1_Init(void)
{
    ADC_ChannelConfTypeDef sConfig = {0};

    hadc1.Instance = ADC1;
    hadc1.Init.ClockPrescaler = ADC_CLOCK_ASYNC_DIV1;
    hadc1.Init.Resolution = ADC_RESOLUTION_12B;
    hadc1.Init.ScanConvMode = ADC_SCAN_DISABLE;
    hadc1.Init.ContinuousConvMode = DISABLE;
    hadc1.Init.DiscontinuousConvMode = DISABLE;
    /* Le déclenchement externe est configuré pour provenir du HRTIM
       La macro ADC_EXTERNALTRIGCONV_HRTIM1_ADC1 est fournie dans l'environnement STM32G4 HAL.
       Vérifiez qu'elle correspond bien à votre mapping de trigger. */
    hadc1.Init.ExternalTrigConv = ADC_EXTERNALTRIGCONV_HRTIM1_ADC1;
    hadc1.Init.ExternalTrigConvEdge = ADC_EXTERNALTRIGCONVEDGE_RISING;
    hadc1.Init.DataAlign = ADC_DATAALIGN_RIGHT;
    hadc1.Init.NbrOfConversion = 1;
    if (HAL_ADC_Init(&hadc1) != HAL_OK)
    {
        Error_Handler();
    }

    /* Configuration du canal ADC à convertir */
    sConfig.Channel = ADC_CHANNEL_1;  // Choisir le canal approprié
    sConfig.Rank = ADC_REGULAR_RANK_1;
    sConfig.SamplingTime = ADC_SAMPLETIME_47CYCLES_5;
    if (HAL_ADC_ConfigChannel(&hadc1, &sConfig) != HAL_OK)
    {
        Error_Handler();
    }
}
