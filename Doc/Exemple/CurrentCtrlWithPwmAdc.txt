#include "stm32g4xx_hal.h"
#include <stdint.h>

/* Définition d'une référence de courant (en unités ADC, à adapter selon votre capteur) */
#define CURRENT_REF 2000  

/* Constantes PID (exemple, à ajuster) */
#define KP 1.0f
#define KI 0.1f
#define KD 0.0f

/* Variables globales */
HRTIM_HandleTypeDef hhrtim1;
ADC_HandleTypeDef hadc1;

volatile uint32_t measuredCurrent = 0;      // Valeur mesurée par l'ADC
volatile float error = 0.0f, error_prev = 0.0f, error_sum = 0.0f;
volatile uint32_t pwmCompareValue = 25000;    // Valeur initiale de comparaison (correspond à 50% de rapport si période=50000)

/* Prototypes de fonctions */
void SystemClock_Config(void);
static void MX_GPIO_Init(void);
static void MX_HRTIM1_Init(void);
static void MX_ADC1_Init(void);
void Error_Handler(void);

/* Callback appelée à la fin d'une conversion ADC */
void HAL_ADC_ConvCpltCallback(ADC_HandleTypeDef *hadc)
{
    if(hadc->Instance == ADC1)
    {
        /* Récupération de la valeur ADC mesurée */
        measuredCurrent = HAL_ADC_GetValue(hadc);

        /* Boucle de régulation PID (simplifiée) */
        error = CURRENT_REF - measuredCurrent;
        error_sum += error;
        float d_error = error - error_prev;
        float pid_output = KP * error + KI * error_sum + KD * d_error;
        error_prev = error;
        
        /* Calcul d'une nouvelle valeur de comparaison pour la PWM */
        int32_t newCompare = (int32_t)pwmCompareValue + (int32_t)pid_output;
        /* Limitation entre 0 et la période maximale (ici 50000) */
        if(newCompare < 0)
            newCompare = 0;
        if(newCompare > 50000)
            newCompare = 50000;
        pwmCompareValue = (uint32_t)newCompare;
        
        /* Mise à jour du registre de comparaison du HRTIM pour le Timer A, Compare 1 */
        HRTIM_CompareCfgTypeDef sCompareCfg = {0};
        sCompareCfg.CompareValue = pwmCompareValue;
        if (HAL_HRTIM_WaveformCompareConfig(&hhrtim1, HRTIM_TIMER_A, HRTIM_COMPARE_1, &sCompareCfg) != HAL_OK)
        {
            Error_Handler();
        }
        
        /* Redémarrage de l'ADC en mode interruption pour la prochaine conversion */
        HAL_ADC_Start_IT(&hadc1);
    }
}

int main(void)
{
    /* Initialisation de la HAL et configuration du système */
    HAL_Init();
    SystemClock_Config();
    MX_GPIO_Init();
    MX_HRTIM1_Init();
    MX_ADC1_Init();
    
    /* Démarrage de la sortie PWM sur le Timer A, Canal 1 */
    if (HAL_HRTIM_WaveformOutputStart(&hhrtim1, HRTIM_OUTPUT_TA1) != HAL_OK)
    {
        Error_Handler();
    }
    if (HAL_HRTIM_WaveformCounterStart(&hhrtim1, HRTIM_TIMER_A) != HAL_OK)
    {
        Error_Handler();
    }
    
    /* Démarrage de l'ADC en mode interruption (déclenché par un trigger externe, ici le HRTIM) */
    if (HAL_ADC_Start_IT(&hadc1) != HAL_OK)
    {
        Error_Handler();
    }
    
    while(1)
    {
        /* Boucle principale : le contrôle se fait en interruption via le callback ADC */
    }
}

/* Configuration des GPIO : ici, on configure par exemple PA8 pour la sortie HRTIM1 (à adapter) */
static void MX_GPIO_Init(void)
{
    __HAL_RCC_GPIOA_CLK_ENABLE();
    GPIO_InitTypeDef GPIO_InitStruct = {0};
    
    /* Exemple : PA8 configurée en mode Alternate Function pour HRTIM1_CHAx */
    GPIO_InitStruct.Pin = GPIO_PIN_8;
    GPIO_InitStruct.Mode = GPIO_MODE_AF_PP;
    GPIO_InitStruct.Pull = GPIO_NOPULL;
    GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH;
    GPIO_InitStruct.Alternate = GPIO_AF13_HRTIM1;  // Vérifier le mapping AF de votre MCU
    HAL_GPIO_Init(GPIOA, &GPIO_InitStruct);
    
    /* Si besoin, configurer la broche pour l'entrée ADC (par exemple, sur une autre broche) */
}

/* Initialisation du HRTIM1 pour générer la PWM */
static void MX_HRTIM1_Init(void)
{
    __HAL_RCC_HRTIM1_CLK_ENABLE();
    
    hhrtim1.Instance = HRTIM1;
    hhrtim1.Init.HRTIMInterruptResquests = HRTIM_IT_NONE;
    hhrtim1.Init.SyncOptions = HRTIM_SYNCOPTION_NONE;
    if (HAL_HRTIM_Init(&hhrtim1) != HAL_OK)
    {
        Error_Handler();
    }
    
    /* Configuration de la base de temps du Timer A */
    HRTIM_TimeBaseCfgTypeDef pTimeBaseCfg = {0};
    pTimeBaseCfg.Period = 50000;          // Période de la PWM (à ajuster selon la fréquence désirée)
    pTimeBaseCfg.RepetitionCounter = 0;
    pTimeBaseCfg.PrescalerRatio = HRTIM_PRESCALERRATIO_DIV1;
    pTimeBaseCfg.Mode = HRTIM_MODE_CONTINUOUS;
    if (HAL_HRTIM_WaveformTimeBaseConfig(&hhrtim1, HRTIM_TIMER_A, &pTimeBaseCfg) != HAL_OK)
    {
        Error_Handler();
    }
    
    /* Configuration du comparateur pour définir la largeur d'impulsion */
    HRTIM_CompareCfgTypeDef sCompareCfg = {0};
    sCompareCfg.CompareValue = pwmCompareValue;
    if (HAL_HRTIM_WaveformCompareConfig(&hhrtim1, HRTIM_TIMER_A, HRTIM_COMPARE_1, &sCompareCfg) != HAL_OK)
    {
        Error_Handler();
    }
    
    /* Configuration de la sortie PWM sur le Timer A, Canal 1 */
    HRTIM_OutputCfgTypeDef pOutputCfg = {0};
    pOutputCfg.Polarity = HRTIM_OUTPUTPOLARITY_HIGH;
    pOutputCfg.IdleMode = HRTIM_OUTPUTIDLEMODE_NONE;
    pOutputCfg.IdleLevel = HRTIM_OUTPUTIDLELEVEL_INACTIVE;
    pOutputCfg.FastMode = HRTIM_OUTPUTFASTMODE_DISABLE;
    pOutputCfg.ChopperModeEnable = HRTIM_OUTPUTCHOPPERMODE_DISABLE;
    pOutputCfg.DutyCycle = 50;  // Valeur initiale (sera ajustée par la régulation)
    if (HAL_HRTIM_WaveformOutputConfig(&hhrtim1, HRTIM_TIMER_A, HRTIM_OUTPUT_TA1, &pOutputCfg) != HAL_OK)
    {
        Error_Handler();
    }
}

/* Initialisation de l'ADC pour mesurer le courant (via la tension sur la résistance de shunt) */
static void MX_ADC1_Init(void)
{
    __HAL_RCC_ADC12_CLK_ENABLE();
    
    hadc1.Instance = ADC1;
    hadc1.Init.ClockPrescaler = ADC_CLOCK_ASYNC_DIV1;
    hadc1.Init.Resolution = ADC_RESOLUTION_12B;
    hadc1.Init.ScanConvMode = ADC_SCAN_DISABLE;
    hadc1.Init.ContinuousConvMode = DISABLE;  // Conversion unique déclenchée par événement
    hadc1.Init.DiscontinuousConvMode = DISABLE;
    hadc1.Init.ExternalTrigConvEdge = ADC_EXTERNALTRIGCONVEDGE_RISING;
    /* Le trigger externe est ici configuré pour provenir du HRTIM (vérifiez la macro correspondant à votre MCU) */
    hadc1.Init.ExternalTrigConv = ADC_EXTERNALTRIGCONV_HRTIM1_ADC1;
    hadc1.Init.DataAlign = ADC_DATAALIGN_RIGHT;
    hadc1.Init.NbrOfConversion = 1;
    if (HAL_ADC_Init(&hadc1) != HAL_OK)
    {
        Error_Handler();
    }
    
    ADC_ChannelConfTypeDef sConfig = {0};
    /* Configuration du canal ADC connecté à la sortie de l'amplificateur de la résistance de shunt */
    sConfig.Channel = ADC_CHANNEL_1;  // À adapter selon votre câblage
    sConfig.Rank = ADC_REGULAR_RANK_1;
    sConfig.SamplingTime = ADC_SAMPLETIME_47CYCLES_5;
    if (HAL_ADC_ConfigChannel(&hadc1, &sConfig) != HAL_OK)
    {
        Error_Handler();
    }
}

/* Fonction de gestion d'erreur */
void Error_Handler(void)
{
    while(1)
    {
        /* Boucle infinie en cas d'erreur */
    }
}
